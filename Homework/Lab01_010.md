#include <stdio.h>

int main() {
    int clearanceLevel, age, isActive;

    if(scanf("%d %d %d", &clearanceLevel, &age, &isActive) != 3) {
        return 1;
    }

    switch (clearanceLevel){
    case 1:
        printf("Access DENIED");
        break;

    case 2:
        if(age >= 25 && isActive == 1) {
            printf("Access Granted");
        } else {
            printf("Access DENIED");
        }
        break;

    case 3:
        if(isActive == 1) {
            printf("Access Granted");
        } else {
            printf("Access DENIED");
        }
        break;

    default:
        printf("Invalid clearance Level");
        break;
    }
    

    return 0;
}
