#include <stdio.h>

int main() {
    float principal, monthlyPayment, interest;
    int mounthCount = 0;
    const float INTEREST_RATE = 0.01;
    const float PENALTY = 10.0;
    int i = 1;

    if(scanf("%f %f", &principal, &monthlyPayment) != 2) {
        return 1;
    }

    while(principal > 0) {
        interest = principal * INTEREST_RATE;
        principal = principal + interest;

        if(monthlyPayment < interest) {
            principal += PENALTY;
        }

        principal = principal - monthlyPayment;
        mounthCount += 1;

        i++;
        if(i <= 10) {
            printf("Month %d:Remaining %.2f\n", i, principal);
        } else {
            break;
        }
    }

    if(i <= 10) {
        printf("Loan settled in %d months.\n", mounthCount);
    } else {
        printf("Loan settled in %d+ months.\n", mounthCount);
    }
    
    return 0;
}
