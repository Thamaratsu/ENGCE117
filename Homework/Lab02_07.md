#include <stdio.h>

int main() {
    int N, i;
    int countFizzBuzz = 0;
    int countFizz = 0;
    int countBuzz = 0;
    int countOther = 0;


    if(scanf("%d", &N) != 1) {
        return 1;
    }

    for(int i = 0; i <= N; i++) {
        
        if(i % 15 == 0) {
            countFizzBuzz++;

        } else if(i % 3 == 0 && i % 5 != 0) {
            countFizz++;

        } else if(i % 5 == 0 && i % 3 != 0) {
            countBuzz++;

        } else {
            countOther++;
        
        }
    }
    

    printf("count FizzBuzz (by 15): %d\n", countFizzBuzz);
    printf("count Fizz (by 3 only): %d\n", countFizz);
    printf("count Buzz (by 5 only): %d\n", countBuzz);
    printf("count Other : %d\n", countOther);
    return 0;
}
