#include <stdio.h>

int main() {
    int numbers[6];
    int even_count = 0;
    int odd_count = 0;
    int i;

    for(i = 0; i < 5; i++) {
        printf("Enter interger number %d: ", i + 1);
        scanf("%d", &numbers[i]);
    }



    printf("\n--- NUMBER CLASSIFICATION REPORT ---\n");
    printf("Recorded Numbers: ");
    for(i = 0; i < 5; i++) {
        printf("%d ", numbers[i]);
        if(numbers[i] % 2 == 0) {
            even_count++;
        } else {
            odd_count++;
        }
    }

    printf("\n");
    printf("Total Even Numbers Found: %d\n", even_count);
    printf("Total Odd Number Found: %d\n", odd_count);

    return 0;
}
