#include <stdio.h>

int main() {
    int MAX_SIZE = 10;
    int data[MAX_SIZE + 1];
    int status[MAX_SIZE + 1];
    int N;
    int i, j;
    int count;


    printf("Enter the number of elements (N, max %d): ", MAX_SIZE);
    scanf("%d", &N);

    if(N > MAX_SIZE || N < 1) {
        N = MAX_SIZE;
    }

    printf("Enter %d integer numbers:\n", N);
    for(i = 0; i < N; i++) {
        printf("Element %d: ", i + 1);
        scanf("%d", &data[i]);
        status[i] = 0;
    }

    printf("\n--- FREQUNECY ANALYSIS REPORT ---\n");
    printf("Total elements recorded (N): %d\n", N);
    printf("Recorded Numbers: ");
    for(i = 0; i < N; i++) {
        printf("%d ", data[i]);
    }

    printf("\n");
    printf("--- FREQUENCY TABLE ---\n");
    printf("Number | Frequency\n");
    printf("-------|----------\n");

    for(i = 0; i < N; i++) {
        if(status[i] == 0) {
            count = 1;

            for(j = i + 1; j < N; j++) {
                if(data[i] == data[j]) {
                    count++;
                    status[j] = -1;
                }
            }

            printf("%6d | %d\n", data[i], count);
        }
    }
    
    return 0;

}
