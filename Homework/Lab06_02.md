## CODE
````c++
#include <stdio.h>

// ฟังก์ชันสลับค่าโดยรับพารามิเตอร์เป็น Pointer เพื่อเข้าถึง Address โดยตรง
void swap_values(int *a, int *b);

int main() {
    int num1 = 0;
    int num2 = 0;

    // รับข้อมูลจากผู้ใช้ผ่าน Standard I/O
    printf("Enter Value for Num1: ");
    if (scanf("%d", &num1) != 1) return 1; // เช็คความถูกต้องของ input

    printf("Enter Value for Num2: ");
    if (scanf("%d", &num2) != 1) return 1;

    printf("\n--- SWAPPING REPORT ---\n");
    printf("Before Swap - Num1: %d, Num2: %d\n", num1, num2);

    // ส่ง Address ของตัวแปร (&) เข้าไปยังฟังก์ชัน
    swap_values(&num1, &num2);

    printf("After Swap - Num1: %d, Num2: %d\n", num1, num2);

    return 0;
}

void swap_values(int *a, int *b) {
    // Check Stability: ตรวจสอบว่า Pointer ไม่เป็นค่าว่างก่อนทำงาน
    if (a == NULL || b == NULL) {
        return; 
    }

    int temp = *a; // เก็บค่าดั้งเดิมของ a ไว้ในตัวแปรชั่วคราว
    *a = *b;       // เปลี่ยนค่าใน Address a ให้เป็นค่าของ b
    *b = temp;     // นำค่าดั้งเดิมของ a มาใส่ใน Address b
}      
````
