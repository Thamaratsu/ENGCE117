#include <stdio.h>
#include <string.h>

#define MAX_BUFFER 100 

int main() {
    char sentence[MAX_BUFFER];
    int vowel_count = 0;
    char *char_ptr = NULL; 

    printf("Enter a sentence: ");
    if (fgets(sentence, MAX_BUFFER, stdin) == NULL) {
        return 1;
    }

    sentence[strcspn(sentence, "\n")] = '\0';

    char_ptr = sentence;

    while (*char_ptr != '\0') {
        char current = *char_ptr;

        if (current == 'a' || current == 'A' ||
            current == 'e' || current == 'E' ||
            current == 'i' || current == 'I' ||
            current == 'o' || current == 'O' ||
            current == 'u' || current == 'U') {
            
            vowel_count++;
        }

        char_ptr++;
    }
    printf("\n--- STRING ANALYSIS REPORT ---\n");
    printf("Input String: \"%s\"\n", sentence);
    printf("Total Vowel Count: %d\n", vowel_count);

    return 0;
    
}
