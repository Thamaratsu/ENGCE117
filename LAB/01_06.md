#include <stdio.h>

int* GetMatrix(int *row, int *col);

int main() {
    int *data, m, n;
    data = GetMatrix(&m, &n);
    
    int total = m * n;
    for(int i = 0; i < total; i++) {
        printf("data[%d] = %d\n", i, data[i]);
    }
    delete[] data;
    return 0;
}

int* GetMatrix(int *row, int *col) {
    int *a;
    printf("Enter Array : \n");
    scanf("%d %d", row, col);

    a = new int[*row * *col];

    for(int i = 0; i < *row; i++) {
        for(int j = 0; j < *col; j++) {
            printf("a[%d][%d] = ", i,j);
            scanf("%d", &a[ i * *col + j]);
        }
    }
    return a;

}
