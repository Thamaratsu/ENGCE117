char* reverse(char str1[]) {
    int length = strlen(str1);
    int end = length - 1;


    for(int start = 0; start < end ; start++, end--) {
        char temp = str1[start];
        str1[start] = str1[end];
        str1[end] = temp;
    }
    return str1;
}
