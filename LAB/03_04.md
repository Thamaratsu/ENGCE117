#include <stdio.h>
struct student {
    char name[20];
    int age;
    char sex;
    float gpa;
};


void GetStudent(struct student child[][10], int *room);

int main() {
    struct student children[20][10];
    int group;
    GetStudent(children, &group);
    return 0;
}

void GetStudent(struct student child[][10], int *room) {
    printf("Room : ");
    scanf("%d", room);

    for(int j = 1; j <= *room; j++) {
        for(int i = 1; i <= 10; i++) {
        printf("group %d Children %d's name, age sex gpa:", j, i);
        scanf("%s %d  %c %f", child[j][i].name,
                            &child[j][i].age,
                            &child[j][i].sex,
                            &child[j][i].gpa);
        }
    }
}
