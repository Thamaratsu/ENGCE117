#include <stdio.h>

struct student {
    char name[20];
    int age;
    char sex;
    float gpa;
};

struct student (*GetStudent(int *room))[10];

int main() {
    struct student(*children)[10];
    int group;
    children = GetStudent(&group);
    return 0;
}

struct student (*GetStudent(int *room))[10] {
    printf("Enter group : ");
    scanf("%d", room);

    struct student (*child)[10];
    child = new struct student[*room][10];
    for(int j = 0; j < *room; j++) {
        for(int i = 0; i < 10; i++) {
        printf("group %d Children %d's name, age sex gpa:", j+1, i+1);
        scanf("%s %d  %c %f", child[j][i].name,
                            &child[j][i].age,
                            &child[j][i].sex,
                            &child[j][i].gpa);
        }
    }
    return child;
}
