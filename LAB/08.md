#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct studentNode {
    char name[20];
    int age;
    char sex;
    float gpa;
    struct studentNode *next;
};

class LinkedList {
    protected:
        struct studentNode *start, **now;
    public:
        LinkedList();
        ~LinkedList();
        void InsNode(char n[], int a, char s, float g);
        void DelNode();
        void GoNext();
        void GoFirst();
        void GoLast();
        void ShowAll();
        int FindNode(char n[]);
        struct studentNode *NowNode();
        void EditNode(char n[], int a, char s, float g);
};

void AddData(LinkedList *ll);
void EditData(LinkedList *ll);
void FindData(LinkedList *ll);
void writefile(LinkedList *ll);
void readfile(LinkedList *ll);

int main() {
    LinkedList listA;
    int menu;

    readfile(&listA);

    printf("\n  Menu - (1) Add  (2) Edit  (3) Delete  (4) Find  (5) Show  (0) Exit : ");
    scanf("%d", &menu);

    while (menu != 0) {
        switch (menu) {
            case 1: AddData(&listA);       break;
            case 2: EditData(&listA);      break;
            case 3: listA.DelNode();       break;
            case 4: FindData(&listA);      break;
            case 5: listA.ShowAll();       break;
            default: printf("  Invalid option.\n");
        }
        printf("\n  Menu - (1) Add  (2) Edit  (3) Delete  (4) Find  (5) Show  (0) Exit : ");
        scanf("%d", &menu);
    }

    writefile(&listA);
    return 0;
}

LinkedList::LinkedList() {
    start = NULL;
    now = &start;
}

LinkedList::~LinkedList() {
    struct studentNode *temp;
    while (start != NULL) {
        temp = start;
        start = start->next;
        free(temp);
    }
}

void LinkedList::InsNode(char n[], int a, char s, float g) {
    struct studentNode *newNode = (struct studentNode*)malloc(sizeof(struct studentNode));
    strcpy(newNode->name, n);
    newNode->age = a;
    newNode->sex = s;
    newNode->gpa = g;
    newNode->next = NULL;

    if (start == NULL) {
        start = newNode;
        now = &start;
    } else {
        GoLast();
        (*now)->next = newNode;
        now = &((*now)->next);
    }
}

void LinkedList::DelNode() {
    if (*now == NULL) return;
    struct studentNode *temp = *now;
    *now = (*now)->next;
    free(temp);
}

void LinkedList::GoNext() {
    if (*now != NULL && (*now)->next != NULL)
        now = &((*now)->next);
}

void LinkedList::GoFirst() {
    now = &start;
}

void LinkedList::GoLast() {
    now = &start;
    while (*now != NULL && (*now)->next != NULL)
        now = &((*now)->next);
}

void LinkedList::ShowAll() {
    struct studentNode *temp = start;
    int i = 0;
    if (temp == NULL) {
        printf("  No data.\n");
        return;
    }
    while (temp != NULL) {
        printf("  %d: %-20s %c %3d %.2f\n", i, temp->name, temp->sex, temp->age, temp->gpa);
        temp = temp->next;
        i++;
    }
}

int LinkedList::FindNode(char n[]) {
    struct studentNode *temp = start;
    int i = 0;
    while (temp != NULL) {
        if (strcmp(temp->name, n) == 0) {
            now = &start;
            for (int j = 0; j < i; j++) now = &((*now)->next);
            return i;
        }
        temp = temp->next;
        i++;
    }
    return -1;
}

struct studentNode* LinkedList::NowNode() {
    return *now;
}

void LinkedList::EditNode(char n[], int a, char s, float g) {
    if (*now == NULL) return;
    strcpy((*now)->name, n);
    (*now)->age = a;
    (*now)->sex = s;
    (*now)->gpa = g;
}

void AddData(LinkedList *ll) {
    char name[20];
    char sex;
    int age;
    float gpa;
    printf("  Enter Name Sex Age GPA: ");
    scanf("%s %c %d %f", name, &sex, &age, &gpa);
    ll->InsNode(name, age, sex, gpa);
    printf("  Added successfully.\n");
}

void EditData(LinkedList *ll) {
    char name[20];
    printf("  Enter name to edit: ");
    scanf("%s", name);
    int idx = ll->FindNode(name);
    if (idx == -1) {
        printf("  Not found.\n");
        return;
    }
    char newName[20];
    char sex;
    int age;
    float gpa;
    printf("  Enter new Name Sex Age GPA: ");
    scanf("%s %c %d %f", newName, &sex, &age, &gpa);
    ll->EditNode(newName, age, sex, gpa);
    printf("  Edited successfully.\n");
}

void FindData(LinkedList *ll) {
    char name[20];
    printf("  Enter name to find: ");
    scanf("%s", name);
    int idx = ll->FindNode(name);
    if (idx == -1) {
        printf("  Not found.\n");
        return;
    }
    struct studentNode *node = ll->NowNode();
    printf("  Found at index %d: %-20s %c %3d %.2f\n",
           idx, node->name, node->sex, node->age, node->gpa);
}

void writefile(LinkedList *ll) {
    FILE *fp = fopen("student.dat", "wb");
    if (fp == NULL) { printf("Can't open file!\n"); return; }

    ll->GoFirst();
    struct studentNode *prev = NULL;
    struct studentNode *node = ll->NowNode();
    while (node != NULL && node != prev) {
        fwrite(node->name, sizeof(char[20]), 1, fp);
        fwrite(&node->sex,  sizeof(char),    1, fp);
        fwrite(&node->age,  sizeof(int),     1, fp);
        fwrite(&node->gpa,  sizeof(float),   1, fp);
        prev = node;
        ll->GoNext();
        node = ll->NowNode();
    }
    fclose(fp);
}

void readfile(LinkedList *ll) {
    FILE *fp = fopen("student.dat", "rb");
    if (fp == NULL) return;

    char name[20];
    char sex;
    int age;
    float gpa;

    for (int i = 0; i < 100; i++) {
        if (fread(name, sizeof(char[20]), 1, fp) == 0) break;
        if (fread(&sex,  sizeof(char),    1, fp) == 0) break;
        if (fread(&age,  sizeof(int),     1, fp) == 0) break;
        if (fread(&gpa,  sizeof(float),   1, fp) == 0) break;
        ll->InsNode(name, age, sex, gpa);
    }
    fclose(fp);
}
