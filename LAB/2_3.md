#include <stdio.h>
#include <string.h>

void explode(const char str1[], char splitter, char str2[][10], int *count);

int main() {
    char out[20][10];
    int num;
    explode("I/Love/You",'/', out, &num);
    
    for(int i = 0; i < num; i++) {
        printf("out[%d] = \"%s\"\n", i, out[i]);
    }
    printf("\ncount = %d", num);

    return 0;
}

void explode(const char str1[], char splitter, char str2[][10], int *count) {
    *count = 0;
    int j = 0;

    const int max_strings = 10;
    const int max_len = 9;

    for(int i = 0; str1[i] != 0; i++) {
        if(str1[i] != splitter && j < max_len) {
            str2[*count][j] = str1[i];
            j++;
        } else {
            str2[*count][j] = '\0';
            (*count)++;
            j = 0;
        }
        
    }

    if(*count < max_strings) {
            str2[*count][j] = '\0';
            (*count)++;
    }
}
